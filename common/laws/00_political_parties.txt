
law_no_political_parties = {
    group = lawgroup_political_parties

	icon = "gfx/interface/icons/law_icons/no_political_parties.dds"

	progressiveness = 0

	on_activate = {
		scripted_effect_parties_disappearence = yes
		activate_law = law_type:law_no_elections
	}

	modifier = {
		country_authority_add = 200
	}

	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_distribution_of_power_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}
	
	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
	
	ai_impose_chance = {
		value = 0
		
		if = {
			limit = {
				has_law = law_type:law_no_political_parties
				OR = {
					AND = {
						power_bloc ?= { has_identity = identity:identity_ideological_union }
						has_strategy = ai_strategy_conservative_agenda					
					}
					has_strategy = ai_strategy_reactionary_agenda
				}
			}
			
			add = base_impose_law_weight
		}
			
		if = {
			limit = {
				scope:target_country = {
					c:BIC ?= ROOT
					OR = {	
						has_journal_entry = je_british_dictate_plantations
						has_journal_entry = je_british_dictate_military
						has_journal_entry = je_british_dictate_universities
					}
				}
			}
			multiply = 0
		}
	}		
}

law_single_party_state = {
    group = lawgroup_political_parties

	icon = "gfx/interface/icons/law_icons/single_party_state.dds"

	progressiveness = 0

	unlocking_technologies = {
		political_agitation
	}

	modifier = {
		country_authority_add = 250
	}

	on_activate = {
		set_only_legal_party_from_ig = ruler.interest_group
		
		custom_tooltip = {
			text = sps_government_reshuffle_tt
			every_interest_group = {
				limit = {
					is_in_government = yes
					NOT = { #Removes all IGs from government that aren't in your party
						has_party = yes
					}
				}
				remove_ruling_interest_group = yes
			}
			every_interest_group = {
				limit = { #Adds all IGs that are in your party to the government
					is_in_government = no
					has_party = yes
				}
				add_ruling_interest_group = yes
			}
		}

		if = {
			limit = {
				has_law = law_type:law_no_elections
