
is_head_of_state = {
	OR = {
		# President / Monarch
		AND = {
			is_ruler = yes
			owner = {
				OR = {
					# President can only be ruler in a republic
					has_law = law_type:law_ruler_head_of_state
					has_law = law_type:law_president_head_of_state
					has_law = law_type:law_shared_head_of_state
				}
			}
		}

		# Prime Minister
		AND = {
			owner = {
				OR = {
					has_law = law_type:law_pm_head_of_state
					has_law = law_type:law_shared_head_of_state
				}
			}
			OR = {
				is_ruler = yes
				owner = {
					any_interest_group = {
						any_scope_character = {
							this ?= root
						}
						is_strongest_ig_in_government = yes
					}
				}
			}
		}
	}
}
