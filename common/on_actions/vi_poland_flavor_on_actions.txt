
on_law_activated = {
	effect = {
		if = {
			limit = {
				has_variable = var_vi_congress_poland_pu
				NOT = { has_law = law_type:law_monarchy }
			}

			trigger_event = vi_POL_events.1
		}
	}
}

on_diplomatic_play_started = {
	effect = {
		if = {
			limit = {
				OR = {
					initiator_is = c:POL
					target_is = c:POL
				}
				has_play_goal = independence
			}

			every_country = {
				limit = { has_variable = var_vi_support_polish_independence }

				save_scope_as = polish_backer

				root = {
					if = {
						limit = {
							initiator_is = c:POL
						}

						add_initiator_backers = {
							scope:polish_backer
						}
					}
					else = {
						add_target_backers = {
							scope:polish_backer
						}
					}
				}
			}
		}
	}
}
